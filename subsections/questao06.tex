\subsection*{Questão 06}

\textit{Quais as unidades de medida do período e frequência?}

\textbf{Resposta:}

\begin{table}[ht]
    \centering
    \small
    \begin{tabular}{lc @{\hspace{1cm}} lc}
        \toprule
        \textbf{Unidade (Período)} & \textbf{Equivalente} & \textbf{Unidade (Frequência)} & \textbf{Equivalente} \\
        \midrule
            Segundos (s)      & $1 \s$    & Hertz (Hz)      & $1 \hz$    \\
            Milissegundos (ms) & $10^{-3} \s$ & Quilohertz (kHz) & $10^{3} \hz$ \\
            Microssegundos ($\us$) & $10^{-6} \s$ & Megahertz (MHz) & $10^{6} \hz$ \\
            Nanossegundos (ns)  & $10^{-9} \s$ & Gigahertz (GHz) & $10^{9} \hz$ \\
            Picossegundos (ps)  & $10^{-12} \s$ & Terahertz (THz) & $10^{12} \hz$ \\
        \bottomrule
    \end{tabular}
\end{table}
